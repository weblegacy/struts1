23  import static org.junit.jupiter.api.Assertions.assertEquals;
24  import static org.junit.jupiter.api.Assertions.assertFalse;
25  import static org.junit.jupiter.api.Assertions.assertTrue;
26  
27  import java.util.Iterator;
28  
29  import org.junit.jupiter.api.AfterEach;
30  import org.junit.jupiter.api.BeforeEach;
31  import org.junit.jupiter.api.Test;
32  
33  /**
34   * Unit tests for the {@link ActionMessages} class.
35   *
36   * @version $Rev$ $Date$
37   */
38  public class TestActionMessages {
39      protected ActionMessages aMsgs = null;
40      protected ActionMessage msg1 = null;
41      protected ActionMessage msg2 = null;
42      protected ActionMessage msg3 = null;
43      protected ActionMessage msg4 = null;
44      protected ActionMessage msg5 = null;
45  
46  
47      @BeforeEach
48      public void setUp() {
49          aMsgs = new ActionMessages();
50  
51          Object[] objs1 = new Object[] { "a", "b", "c", "d", "e" };
52          Object[] objs2 = new Object[] { "f", "g", "h", "i", "j" };
53  
54          msg1 = new ActionMessage("aMessage", objs1);
55          msg2 = new ActionMessage("anMessage", objs2);
56          msg3 = new ActionMessage("msg3", "value1");
57          msg4 = new ActionMessage("msg4", "value2");
58          msg5 = new ActionMessage("msg5", "value3", "value4");
59      }
60  
61      @AfterEach
62      public void tearDown() {
63          aMsgs = null;
64      }
65  
66      @Test
67      public void testEmpty() {
68          assertTrue(aMsgs.isEmpty(), "aMsgs is not empty!");
69      }
70  
71      @Test
72      public void testNotEmpty() {
73          aMsgs.add("myProp", msg1);
74          assertFalse(aMsgs.isEmpty(), "aMsgs is empty!");
75      }
76  
77      @Test
78      public void testSizeWithOneProperty() {
79          aMsgs.add("myProp", msg1);
80          aMsgs.add("myProp", msg2);
81          assertEquals(2, aMsgs.size("myProp"), "number of mesages is not 2");
82      }
83  
84      @Test
85      public void testSizeWithManyProperties() {
86          aMsgs.add("myProp1", msg1);
87          aMsgs.add("myProp2", msg2);
88          aMsgs.add("myProp3", msg3);
89          aMsgs.add("myProp3", msg4);
90          aMsgs.add("myProp4", msg5);
91          assertEquals(1, aMsgs.size("myProp1"),
92              "number of messages for myProp1 is not 1");
93          assertEquals(5, aMsgs.size(), "number of messages");
94      }
95  
96      @Test
97      public void testSizeAndEmptyAfterClear() {
98          testSizeWithOneProperty();
99          aMsgs.clear();
100         testEmpty();
101         assertEquals(0, aMsgs.size("myProp"), "number of meesages is not 0");
102     }
103 
104     @Test
105     public void testGetWithNoProperty() {
106         Iterator<ActionMessage> it = aMsgs.get("myProp");
107 
108         assertFalse(it.hasNext(), "iterator is not empty!");
109     }
110 
111     @Test
112     public void testGetForAProperty() {
113         testSizeWithOneProperty();
114 
115         Iterator<ActionMessage> it = aMsgs.get("myProp");
116 
117         assertTrue(it.hasNext(), "iterator is empty!");
118     }
119 
120     /**
121      * Tests adding an ActionMessages object to an ActionMessages object.
122      */
123     @Test
124     public void testAddMessages() {
125         ActionMessage msg1 = new ActionMessage("key");
126         ActionMessage msg2 = new ActionMessage("key2");
127         ActionMessage msg3 = new ActionMessage("key3");
128         ActionMessages msgs = new ActionMessages();
129         ActionMessages add = new ActionMessages();
130 
131         msgs.add("prop1", msg1);
132         add.add("prop1", msg2);
133         add.add("prop3", msg3);
134 
135         msgs.add(add);
136         assertEquals(3, msgs.size());
137         assertEquals(2, msgs.size("prop1"));
138 
139         // test message order
140         Iterator<ActionMessage> props = msgs.get();
141         int count = 1;
142 
143         while (props.hasNext()) {
144             ActionMessage msg = props.next();
145 
146             if (count == 1) {
147                 assertEquals("key", msg.getKey());
148             } else if (count == 2) {
149                 assertEquals("key2", msg.getKey());
150             } else {
151                 assertEquals("key3", msg.getKey());
152             }
153 
154             count++;
155         }
156     }
