137     // Test initialize() method on indexed values to ensure that the
138     // result returned by FormPropertyConfig().initial() is never clobbered
139     @Test
140     public void testIndexedInitialize() {
141         // Update some values in the indexed properties
142         dynaForm.set("intArray", 1, Integer.valueOf(111));
143         assertEquals(Integer.valueOf(111), dynaForm.get("intArray", 1),
144             "intArray[1]");
145         dynaForm.set("intIndexed", 2, Integer.valueOf(222));
146         assertEquals(Integer.valueOf(222), dynaForm.get("intIndexed", 2),
147             "intIndexed[2]");
148         dynaForm.set("stringArray", 3, "New String 3");
149         assertEquals("New String 3", dynaForm.get("stringArray", 3),
150             "stringArray[3]");
151         dynaForm.set("stringIndexed", 4, "New String 4");
152         assertEquals("New String 4", dynaForm.get("stringIndexed", 4),
153             "stringIndexed[4]");
154 
155         // Perform initialize() and revalidate the original values
156         // while ensuring our initial values did not get corrupted
157         dynaForm.initialize(mapping);
158         setupComplexProperties();
159         testGetIndexedValues();
160     }

270     /**
271      * Positive test for getDynaPropertys().  Each property name listed in
272      * <code>properties</code> should be returned exactly once.
273      */
274     @Test
275     public void testGetDescriptors() {
276         DynaProperty[] pd = dynaForm.getDynaClass().getDynaProperties();
277 
278         assertNotNull(pd, "Got descriptors");
279 
280         int[] count = new int[properties.length];
281 
282         for (int i = 0; i < pd.length; i++) {
283             String name = pd[i].getName();
284 
285             for (int j = 0; j < properties.length; j++) {
286                 if (name.equals(properties[j])) {
287                     count[j]++;
288                 }
289             }
290         }
291 
292         for (int j = 0; j < properties.length; j++) {
293             if (count[j] < 0) {
294                 fail("Missing property " + properties[j]);
295             } else if (count[j] > 1) {
296                 fail("Duplicate property " + properties[j]);
297             }
298         }
299     }
