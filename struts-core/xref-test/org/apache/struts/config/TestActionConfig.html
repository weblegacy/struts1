203     /**
204      * Test a typical form bean configuration extension where various forwards
205      * and exception handlers should be inherited from a base form. This
206      * method checks all the subelements.
207      */
208     @Test
209     public void testInheritFrom()
210         throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException {
211         // create a basic subform
212         ActionConfig subConfig = new ActionConfig();
213         String subConfigPath = "subConfig";
214 
215         subConfig.setPath(subConfigPath);
216         subConfig.setExtends("/base");
217 
218         // override success
219         ForwardConfig forward = new ForwardConfig();
220 
221         forward.setName("success");
222         forward.setPath("/newSuccess.jsp");
223         forward.setRedirect(true);
224         subConfig.addForwardConfig(forward);
225 
226         // add an exception handler
227         ExceptionConfig handler = new ExceptionConfig();
228 
229         handler.setType("java.lang.NullPointerException");
230         handler.setKey("msg.exception.npe");
231         subConfig.addExceptionConfig(handler);
232 
233         // override arbitrary "label" property
234         subConfig.setProperty("label", "sub");
235 
236         config.addActionConfig(subConfig);
237 
238         subConfig.inheritFrom(baseConfig);
239 
240         // check that our subConfig is still the one in the config
241         assertSame(subConfig, config.findActionConfig("subConfig"),
242             "subConfig no longer in ModuleConfig");
243 
244         // check our configured sub config
245         assertNotNull(subConfig.getType(), "Action type was not inherited");
246         assertEquals(subConfigPath, subConfig.getPath(), "Wrong config path");
247         assertEquals(baseConfig.getType(), subConfig.getType(),
248             "Wrong config type");
249 
250         // check our forwards
251         ForwardConfig[] forwards = subConfig.findForwardConfigs();
252 
253         assertEquals(2, forwards.length, "Wrong forwards count");
254 
255         forward = subConfig.findForwardConfig("success");
256         assertNotNull(forward, "'success' forward was not found");
257         assertEquals("/newSuccess.jsp", forward.getPath(),
258             "Wrong path for success");
259 
260         forward = subConfig.findForwardConfig("failure");
261 
262         ForwardConfig origForward = baseConfig.findForwardConfig("failure");
263 
264         assertNotNull(forward, "'failure' forward was not inherited");
265         assertEquals(origForward.getPath(), forward.getPath(),
266             "Wrong type for 'failure'");
267         assertEquals(origForward.getProperty("forwardCount"),
268             forward.getProperty("forwardCount"),
269             "Arbitrary property not copied");
270 
271         // check our exceptions
272         ExceptionConfig[] handlers = subConfig.findExceptionConfigs();
273 
274         assertEquals(2, handlers.length, "Wrong exception config count");
275 
276         handler = subConfig.findExceptionConfig("java.sql.SQLException");
277 
278         ExceptionConfig origHandler =
279             baseConfig.findExceptionConfig("java.sql.SQLException");
280 
281         assertNotNull(handler, "'SQLException' handler was not found");
282         assertEquals(origHandler.getKey(), handler.getKey(),
283             "Wrong key for 'SQLException'");
284         assertEquals(origHandler.getProperty("exceptionCount"),
285             handler.getProperty("exceptionCount"),
286             "Arbitrary property not copied");
287 
288         handler =
289             subConfig.findExceptionConfig("java.lang.NullPointerException");
290         assertNotNull(handler, "'NullPointerException' handler disappeared");
291 
292         // check the arbitrary properties
293         String version = subConfig.getProperty("version");
294 
295         assertEquals("1a", version,
296             "Arbitrary property 'version' wasn't inherited");
297 
298         String label = subConfig.getProperty("label");
299 
300         assertEquals("sub", label,
301             "Arbitrary property 'label' shouldn't have changed");
302     }

304     /**
305      * Make sure that correct exception is thrown if a base action can't be
306      * found.
307      */
308     @Test
309     public void testInheritBoolean()
310         throws ClassNotFoundException, IllegalAccessException, InstantiationException, InvocationTargetException {
311 
312         ActionConfig parentConfig = new ActionConfig();
313         parentConfig.setPath("/parent");
314         ActionConfig childConfig  = null;
315 
316         // Test if boolean is NOT set it IS inherited
317         parentConfig.setValidate(true);
318         parentConfig.setCancellable(true);
319         childConfig = new ActionConfig();
320         childConfig.inheritFrom(parentConfig);
321         assertTrue(childConfig.getValidate(), "default validate inherit true");
322         assertTrue(childConfig.getCancellable(), "default cancellable inherit true");
323 
324         // Test if boolean is NOT set it IS inherited
325         parentConfig.setValidate(false);
326         parentConfig.setCancellable(false);
327         childConfig = new ActionConfig();
328         childConfig.inheritFrom(parentConfig);
329         assertFalse(childConfig.getValidate(), "default validate inherit false");
330         assertFalse(childConfig.getCancellable(), "default cancellable inherit false");
331 
332         // Test if boolean IS set it is NOT inherited
333         parentConfig.setValidate(true);
334         parentConfig.setCancellable(true);
335         childConfig = new ActionConfig();
336         childConfig.setValidate(false);
337         childConfig.setCancellable(false);
338         childConfig.inheritFrom(parentConfig);
339         assertFalse(childConfig.getValidate(), "set validate (not inherit true)");
340         assertFalse(childConfig.getCancellable(), "set cancellable (not inherit false)");
341 
342         // Test if boolean IS set it is NOT inherited
343         parentConfig.setValidate(false);
344         parentConfig.setCancellable(false);
345         childConfig = new ActionConfig();
346         childConfig.setValidate(true);
347         childConfig.setCancellable(true);
348         childConfig.inheritFrom(parentConfig);
349         assertTrue(childConfig.getValidate(), "set validate (not inherit false)");
350         assertTrue(childConfig.getCancellable(), "set cancellable (not inherit false)");
351 
352     }
