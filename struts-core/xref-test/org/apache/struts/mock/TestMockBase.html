39  /**
40   * Convenience base class for unit tests of the {@code org.apache.struts.mock}
41   * package, and others that require a runtime environment similar to what the
42   * Struts controller servlet sets up.  The {@link #setUp()} method
43   * establishes a consistent basic environment for the various tests.  The only
44   * tests included in this class are simple validations that the basic
45   * environment was set up correctly.
46   *
47   * @version $Rev$ $Date$
48   */
49  public class TestMockBase {
50      // ----------------------------------------------------- Instance Variables
51      protected ModuleConfig moduleConfig = null;
52      protected ModuleConfig moduleConfig2 = null;
53      protected ModuleConfig moduleConfig3 = null;
54      protected MockServletConfig config = null;
55      protected MockServletContext context = null;
56      protected MockPageContext page = null;
57      protected MockPrincipal principal = null;
58      protected MockHttpServletRequest request = null;
59      protected MockHttpServletResponse response = null;
60      protected MockHttpSession session = null;
61  
62      // ----------------------------------------------------- Setup and Teardown
63      @BeforeEach
64      public void setUp() {
65          // Set up the servlet API objects for a test scenario
66          context = new MockServletContext();
67          config = new MockServletConfig(context);
68          session = new MockHttpSession(context);
69          request = new MockHttpServletRequest(session);
70          principal =
71              new MockPrincipal("username", new String[] { "admin", "manager" });
72          request.setUserPrincipal(principal);
73          response = new MockHttpServletResponse();
74          page = new MockPageContext(config, request, response);
75  
76          // Set up application configurations for our supported modules
77          setUpDefaultApp();
78          setUpSecondApp();
79          setUpThirdApp();
80  
81          // NOTE - we do not initialize the request attribute
82          // for the selected module so that fallbacks to the
83          // default module can be tested.  To select a module,
84          // tests should set the request attribute Globals.MODULE_KEY
85          // to the ModuleConfig instance for the selected module
86      }
87  
88      protected void setUpDefaultApp() {
89          ActionFormBean formBean = null;
90          ActionMapping mapping = null;
91  
92          ModuleConfigFactory factoryObject = ModuleConfigFactory.createFactory();
93  
94          moduleConfig = factoryObject.createModuleConfig("");
95  
96          context.setAttribute(Globals.MODULE_KEY, moduleConfig);
97  
98          // Forward "external" to "http://jakarta.apache.org/"
99          moduleConfig.addForwardConfig(new ActionForward("external",
100                 "http://jakarta.apache.org/", false));
101 
102         // Forward "foo" to "/bar.jsp"
103         moduleConfig.addForwardConfig(new ActionForward("foo", "/bar.jsp", false));
104 
105         // Forward "relative1" to "relative.jsp" non-context-relative
106         moduleConfig.addForwardConfig(new ActionForward("relative1",
107                 "relative.jsp", false));
108 
109         // Forward "relative2" to "relative.jsp" context-relative
110         moduleConfig.addForwardConfig(new ActionForward("relative2",
111                 "relative.jsp", false));
112 
113         // Form Bean "static" is a standard ActionForm subclass
114         formBean =
115             new ActionFormBean("static", "org.apache.struts.mock.MockFormBean");
116         moduleConfig.addFormBeanConfig(formBean);
117 
118         // Action "/static" uses the "static" form bean in request scope
119         mapping = new ActionMapping();
120         mapping.setInput("/static.jsp");
121         mapping.setName("static");
122         mapping.setPath("/static");
123         mapping.setScope("request");
124         mapping.setType("org.apache.struts.mock.MockAction");
125         moduleConfig.addActionConfig(mapping);
126 
127         // Form Bean "dynamic" is a DynaActionForm with the same properties
128         formBean =
129             new ActionFormBean("dynamic",
130                 "org.apache.struts.action.DynaActionForm");
131         formBean.addFormPropertyConfig(new FormPropertyConfig(
132                 "booleanProperty", "boolean", "false"));
133         formBean.addFormPropertyConfig(new FormPropertyConfig(
134                 "stringProperty", "java.lang.String", null));
135         moduleConfig.addFormBeanConfig(formBean);
136 
137         // Action "/dynamic" uses the "dynamic" form bean in session scope
138         mapping = new ActionMapping();
139         mapping.setInput("/dynamic.jsp");
140         mapping.setName("dynamic");
141         mapping.setPath("/dynamic");
142         mapping.setScope("session");
143         mapping.setType("org.apache.struts.mock.MockAction");
144         moduleConfig.addActionConfig(mapping);
145 
146         // Form Bean "/dynamic0" is a DynaActionForm with initializers
147         formBean =
148             new ActionFormBean("dynamic0",
149                 "org.apache.struts.action.DynaActionForm");
150         formBean.addFormPropertyConfig(new FormPropertyConfig(
151                 "booleanProperty", "boolean", "true"));
152         formBean.addFormPropertyConfig(new FormPropertyConfig(
153                 "stringProperty", "java.lang.String", "String Property"));
154         formBean.addFormPropertyConfig(new FormPropertyConfig("intArray1",
155                 "int[]", "{1,2,3}", 4)); // 4 should be ignored
156         formBean.addFormPropertyConfig(new FormPropertyConfig("intArray2",
157                 "int[]", null, 5)); // 5 should be respected
158         formBean.addFormPropertyConfig(new FormPropertyConfig("principal",
159                 "org.apache.struts.mock.MockPrincipal", null));
160         formBean.addFormPropertyConfig(new FormPropertyConfig("stringArray1",
161                 "java.lang.String[]", "{aaa,bbb,ccc}", 2)); // 2 should be ignored
162         formBean.addFormPropertyConfig(new FormPropertyConfig("stringArray2",
163                 "java.lang.String[]", null, 3)); // 3 should be respected
164         moduleConfig.addFormBeanConfig(formBean);
165 
166         // Action "/dynamic0" uses the "dynamic0" form bean in request scope
167         mapping = new ActionMapping();
168         mapping.setName("dynamic0");
169         mapping.setPath("/dynamic0");
170         mapping.setScope("request");
171         mapping.setType("org.apache.struts.mock.MockAction");
172         moduleConfig.addActionConfig(mapping);
173 
174         // Action "/noform" has no form bean associated with it
175         mapping = new ActionMapping();
176         mapping.setPath("/noform");
177         mapping.setType("org.apache.struts.mock.MockAction");
178         moduleConfig.addActionConfig(mapping);
179 
180         // Configure global forward declarations
181         moduleConfig.addForwardConfig(new ForwardConfig("moduleForward",
182                 "/module/forward", false)); // No redirect, same module
183 
184         moduleConfig.addForwardConfig(new ForwardConfig("moduleRedirect",
185                 "/module/redirect", true)); // Redirect, same module
186 
187         moduleConfig.addForwardConfig(new ForwardConfig("contextForward",
188                 "/forward", false, // No redirect
189                 "/context")); // Specify module
190 
191         moduleConfig.addForwardConfig(new ForwardConfig("contextRedirect",
192                 "/redirect", true, // Redirect
193                 "/context")); // Specify module
194 
195         moduleConfig.addForwardConfig(new ForwardConfig("moduleNoslash",
196                 "module/noslash", false)); // No redirect, same module
197 
198         moduleConfig.addForwardConfig(new ForwardConfig("contextNoslash",
199                 "noslash", false, // No redirect
200                 "/context")); // Specify module
201     }

291     // Set up third app for testing URL mapping
292     protected void setUpThirdApp() {
293         ModuleConfigFactory factoryObject = ModuleConfigFactory.createFactory();
294 
295         moduleConfig3 = factoryObject.createModuleConfig("/3");
296 
297         context.setAttribute(Globals.MODULE_KEY + "/3", moduleConfig3);
298 
299         // Instantiate the controller configuration for this app
300         ControllerConfig controller = new ControllerConfig();
301 
302         moduleConfig3.setControllerConfig(controller);
303 
304         // Configure the properties we will be testing
305         controller.setForwardPattern("/forwarding$M$P");
306         controller.setInputForward(true);
307         controller.setPagePattern("/paging$M$P");
308 
309         // Configure global forward declarations
310         moduleConfig3.addForwardConfig(new ForwardConfig("moduleForward",
311                 "/module/forward", false)); // No redirect, same module
312 
313         moduleConfig3.addForwardConfig(new ForwardConfig("moduleRedirect",
314                 "/module/redirect", true)); // Redirect, same module
315 
316         moduleConfig3.addForwardConfig(new ForwardConfig("contextForward",
317                 "/forward", false, // No redirect
318                 "/context")); // Specify module
319 
320         moduleConfig3.addForwardConfig(new ForwardConfig("contextRedirect",
321                 "/redirect", true, // Redirect
322                 "/context")); // Specify module
323 
324         moduleConfig3.addForwardConfig(new ForwardConfig("moduleNoslash",
325                 "module/noslash", false)); // No redirect, same module
326 
327         moduleConfig3.addForwardConfig(new ForwardConfig("contextNoslash",

346     // ------------------------------------------------------- Individual Tests
347     // Test the basic setup of the mock object environment
348     @Test
349     public void testUtilBaseEnvironment() {
350         // Validate the servlet API objects and inter-relationships
351         assertNotNull(config, "config is present");
352         assertNotNull(context, "context is present");
353         assertNotNull(page, "page is present");
354         assertNotNull(principal, "principal is present");
355         assertNotNull(request, "request is present");
356         assertNotNull(response, "response is present");
357         assertNotNull(session, "session is present");
358         assertEquals(config, page.getServletConfig(), "page-->config");
359         assertEquals(context, config.getServletContext(), "config-->context");
360         assertEquals(context, page.getServletContext(), "page-->context");
361         assertEquals(request, page.getRequest(), "page-->request");
362         assertEquals(response, page.getResponse(), "page-->response");
363         assertEquals(session, page.getSession(), "page-->session");
364         assertEquals(principal, request.getUserPrincipal(),
365             "request-->principal");
366         assertEquals(session, request.getSession(), "request-->session");
367         assertEquals(context, session.getServletContext(), "session-->context");
368 
369         // Validate the configuration for the default module
370         assertNotNull(moduleConfig, "moduleConfig is present");
371         assertEquals(moduleConfig, context.getAttribute(Globals.MODULE_KEY),
372             "context-->moduleConfig");
373 
374         // Validate the configuration for the second module
375         assertNotNull(moduleConfig2, "moduleConfig2 is present");
376         assertEquals(moduleConfig2, context.getAttribute(Globals.MODULE_KEY + "/2"),
377             "context-->moduleConfig2");
378     }
