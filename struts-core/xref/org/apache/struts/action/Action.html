39  /**
40   * <p>An <strong>Action</strong> is an adapter between the contents of an
41   * incoming HTTP request and the corresponding business logic that should be
42   * executed to process this request. The controller (RequestProcessor) will
43   * select an appropriate Action for each request, create an instance (if
44   * necessary), and call the <code>execute</code> method.</p>
45   *
46   * <p>Actions must be programmed in a thread-safe manner, because the
47   * controller will share the same instance for multiple simultaneous requests.
48   * This means you should design with the following items in mind: </p>
49   *
50   * <ul>
51   *
52   * <li>Instance and static variables MUST NOT be used to store information
53   * related to the state of a particular request. They MAY be used to share
54   * global resources across requests for the same action.</li>
55   *
56   * <li>Access to other resources (JavaBeans, session variables, etc.) MUST be
57   * synchronized if those resources require protection. (Generally, however,
58   * resource classes should be designed to provide their own protection where
59   * necessary.</li>
60   *
61   * </ul>
62   *
63   * <p>When an <code>Action</code> instance is first created, the controller
64   * will call <code>setServlet</code> with a non-null argument to identify the
65   * servlet instance to which this Action is attached. When the servlet is to
66   * be shut down (or restarted), the <code>setServlet</code> method will be
67   * called with a <code>null</code> argument, which can be used to clean up any
68   * allocated resources in use by this Action.</p>
69   *
70   * @version $Rev$ $Date: 2005-08-26 21:58:39 -0400 (Fri, 26 Aug 2005)
71   *          $
72   */

152     /**
153      * <p>Process the specified non-HTTP request, and create the corresponding
154      * non-HTTP response (or forward to another web component that will create
155      * it), with provision for handling exceptions thrown by the business
156      * logic. Return an {@link ActionForward} instance describing where and
157      * how control should be forwarded, or <code>null</code> if the response
158      * has already been completed.</p>
159      *
160      * <p>The default implementation attempts to forward to the HTTP version
161      * of this method.</p>
162      *
163      * @param mapping  The ActionMapping used to select this instance
164      * @param form     The optional ActionForm bean for this request (if any)
165      * @param request  The non-HTTP request we are processing
166      * @param response The non-HTTP response we are creating
167      * @return The forward to which control should be transferred, or
168      *         <code>null</code> if the response has been completed.
169      * @throws Exception if the application business logic throws an
170      *                   exception.
171      * @since Struts 1.1
172      */
173     public ActionForward execute(ActionMapping mapping, ActionForm form,
174         ServletRequest request, ServletResponse response)
175         throws Exception {
176         try {
177             return execute(mapping, form, (HttpServletRequest) request,
178                 (HttpServletResponse) response);
179         } catch (ClassCastException e) {
180             return null;
181         }
182     }

210     /**
211      * Adds the specified messages keys into the appropriate request attribute
212      * for use by the &lt;html:messages&gt; tag (if messages="true" is set),
213      * if any messages are required. Initialize the attribute if it has not
214      * already been. Otherwise, ensure that the request attribute is not set.
215      *
216      * @param request  The servlet request we are processing
217      * @param messages Messages object
218      * @since Struts 1.2.1
219      */
220     protected void addMessages(HttpServletRequest request,
221         ActionMessages messages) {
222         if (messages == null) {
223             //  bad programmer! *slap*
224             return;
225         }
226 
227         // get any existing messages from the request, or make a new one
228         ActionMessages requestMessages =
229             (ActionMessages) request.getAttribute(Globals.MESSAGE_KEY);
230 
231         if (requestMessages == null) {
232             requestMessages = new ActionMessages();
233         }
234 
235         // add incoming messages
236         requestMessages.add(messages);
237 
238         // if still empty, just wipe it out from the request
239         if (requestMessages.isEmpty()) {
240             request.removeAttribute(Globals.MESSAGE_KEY);
241 
242             return;
243         }
244 
245         // Save the messages
246         request.setAttribute(Globals.MESSAGE_KEY, requestMessages);
247     }

253      * that the request attribute is not set.
254      *
255      * @param request The servlet request we are processing
256      * @param errors  Errors object
257      * @since Struts 1.2.1

267             (ActionMessages) request.getAttribute(Globals.ERROR_KEY);
268 
269         if (requestErrors == null) {
270             requestErrors = new ActionMessages();

275 
276         // if still empty, just wipe it out from the request

363     /**
364      * <p>Return the specified message resources for the current module.</p>
365      *
366      * @param request The servlet request we are processing
367      * @param key     The key specified in the message-resources element for
368      *                the requested bundle.
369      * @return The specified message resource for the current module.
370      * @since Struts 1.1
371      */
372     protected MessageResources getResources(HttpServletRequest request,
373         String key) {
374         // Identify the current module
375         ServletContext context = getServlet().getServletContext();
376         ModuleConfig moduleConfig =
377             ModuleUtils.getInstance().getModuleConfig(request, context);
378 
379         // Return the requested message resources instance
380         return (MessageResources) context.getAttribute(key
381             + moduleConfig.getPrefix());
382     }

384     /**
385      * <p>Returns <code>true</code> if the current form's cancel button was
386      * pressed. This method will check if the <code>Globals.CANCEL_KEY</code>
387      * request attribute has been set, which normally occurs if the cancel
388      * button generated by <strong>CancelTag</strong> was pressed by the user
389      * in the current request. If <code>true</code>, validation performed by
390      * an <strong>ActionForm</strong>'s <code>validate()</code> method will
391      * have been skipped by the controller servlet.</p>
392      *
393      * <p> Since Action 1.3.0, the mapping for a cancellable Action must also have
394      * the new "cancellable" property set to true. If "cancellable" is not set, and
395      * the magic Cancel token is found in the request, the standard Composable
396      * Request Processor will throw an InvalidCancelException. </p>
397      *
398      * @param request The servlet request we are processing
399      * @return <code>true</code> if the cancel button was pressed;
400      *         <code>false</code> otherwise.
401      */
402     protected boolean isCancelled(HttpServletRequest request) {
403         return (request.getAttribute(Globals.CANCEL_KEY) != null);
404     }
