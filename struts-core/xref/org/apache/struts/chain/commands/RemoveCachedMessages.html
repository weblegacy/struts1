21  package org.apache.struts.chain.commands;
22  
23  import java.util.Map;
24  import org.apache.struts.Globals;
25  import org.apache.struts.chain.contexts.ActionContext;
26  import org.apache.struts.action.ActionMessages;
27  
28  /**
29   * <p>Remove cached messages stored in the session.</p>
30   *
31   * @version $Id$
32   * @see CacheMessages
33   * @since Struts 1.3.5
34   */
35  public class RemoveCachedMessages extends ActionCommandBase {
36  
37      /**
38       * <p>Removes any <code>ActionMessages</code> object stored in the session
39       * under <code>Globals.MESSAGE_KEY</code> and <code>Globals.ERROR_KEY</code>
40       * if the messages' <code>isAccessed</code> method returns true.  This
41       * allows messages to be stored in the session, displayed one time, and be
42       * released here.</p>
43       *
44       * @param actionCtx The <code>Context</code> for the current request
45       * @return <code>false</code> so that processing continues
46       * @throws Exception on any error
47       */
48      public boolean execute(ActionContext actionCtx)
49          throws Exception {
50  
51          // Get session scope
52          Map session = actionCtx.getSessionScope();
53  
54          // Remove messages as needed
55          removeAccessedMessages(session, Globals.MESSAGE_KEY);
56  
57          // Remove error messages as needed
58          removeAccessedMessages(session, Globals.ERROR_KEY);
59  
60          return CONTINUE_PROCESSING;
61      }
62  
63      /**
64       * <p>Removes any <code>ActionMessages</code> object from the specified
65       * scope stored under the specified key if the messages'
66       * <code>isAccessed</code> method returns true.
67       *
68       * @param scope The scope to check for messages in.
69       * @param key The key the messages are stored under.
70       */
71      private void removeAccessedMessages(Map scope, String key) {
72          ActionMessages messages = (ActionMessages)scope.get(key);
73          if (messages != null && messages.isAccessed()) {
74              scope.remove(key);
75          }
76      }
77  }
