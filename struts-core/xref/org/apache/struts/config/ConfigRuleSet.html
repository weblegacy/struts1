31  /**
32   * <p>The set of Digester rules required to parse a Struts configuration file
33   * (<code>struts-config.xml</code>).</p>
34   *
35   * @version $Rev$ $Date: 2005-08-16 15:53:27 -0400 (Tue, 16 Aug 2005)
36   *          $
37   * @since Struts 1.1

39  public class ConfigRuleSet extends RuleSetBase {
40      // --------------------------------------------------------- Public Methods
41  
42      /**
43       * <p>Add the set of Rule instances defined in this RuleSet to the
44       * specified <code>Digester</code> instance, associating them with our
45       * namespace URI (if any).  This method should only be called by a
46       * Digester instance.  These rules assume that an instance of
47       * <code>org.apache.struts.config.ModuleConfig</code> is pushed onto the
48       * evaluation stack before parsing begins.</p>
49       *
50       * @param digester Digester instance to which the new Rule instances
51       *                 should be added.
52       */
53      public void addRuleInstances(Digester digester) {
54          ClassLoader cl = digester.getClassLoader();
55  
56          digester.addRule("struts-config/set-property",
57              new BaseConfigSetPropertyRule());
58  
59          digester.addRule("struts-config/action-mappings",
60              new SetActionMappingClassRule());
61  
62          digester.addFactoryCreate("struts-config/action-mappings/action",
63              new ActionMappingFactory(cl));
64          digester.addSetProperties("struts-config/action-mappings/action");
65          digester.addSetNext("struts-config/action-mappings/action",
66              "addActionConfig", "org.apache.struts.config.ActionConfig");
67  
68          digester.addRule("struts-config/action-mappings/action/set-property",
69              new BaseConfigSetPropertyRule());
70  
71          digester.addObjectCreate("struts-config/action-mappings/action/exception",
72              "org.apache.struts.config.ExceptionConfig", "className");
73          digester.addSetProperties(
74              "struts-config/action-mappings/action/exception");
75          digester.addSetNext("struts-config/action-mappings/action/exception",
76              "addExceptionConfig", "org.apache.struts.config.ExceptionConfig");
77  
78          digester.addRule("struts-config/action-mappings/action/exception/set-property",
79              new BaseConfigSetPropertyRule());
80  
81          digester.addFactoryCreate("struts-config/action-mappings/action/forward",
82              new ActionForwardFactory(cl));
83          digester.addSetProperties(
84              "struts-config/action-mappings/action/forward");
85          digester.addSetNext("struts-config/action-mappings/action/forward",
86              "addForwardConfig", "org.apache.struts.config.ForwardConfig");
87  
88          digester.addRule("struts-config/action-mappings/action/forward/set-property",
89              new BaseConfigSetPropertyRule());
90  
91          digester.addObjectCreate("struts-config/controller",
92              "org.apache.struts.config.ControllerConfig", "className");
93          digester.addSetProperties("struts-config/controller");
94          digester.addSetNext("struts-config/controller", "setControllerConfig",
95              "org.apache.struts.config.ControllerConfig");
96  
97          digester.addRule("struts-config/controller/set-property",
98              new BaseConfigSetPropertyRule());
99  
100         digester.addRule("struts-config/form-beans",
101             new SetActionFormBeanClassRule());
102 
103         digester.addFactoryCreate("struts-config/form-beans/form-bean",
104             new ActionFormBeanFactory(cl));
105         digester.addSetProperties("struts-config/form-beans/form-bean");
106         digester.addSetNext("struts-config/form-beans/form-bean",
107             "addFormBeanConfig", "org.apache.struts.config.FormBeanConfig");
108 
109         digester.addObjectCreate("struts-config/form-beans/form-bean/form-property",
110             "org.apache.struts.config.FormPropertyConfig", "className");
111         digester.addSetProperties(
112             "struts-config/form-beans/form-bean/form-property");
113         digester.addSetNext("struts-config/form-beans/form-bean/form-property",
114             "addFormPropertyConfig",
115             "org.apache.struts.config.FormPropertyConfig");
116 
117         digester.addRule("struts-config/form-beans/form-bean/form-property/set-property",
118             new BaseConfigSetPropertyRule());
119 
120         digester.addRule("struts-config/form-beans/form-bean/set-property",
121             new BaseConfigSetPropertyRule());
122 
123         digester.addObjectCreate("struts-config/global-exceptions/exception",
124             "org.apache.struts.config.ExceptionConfig", "className");
125         digester.addSetProperties("struts-config/global-exceptions/exception");
126         digester.addSetNext("struts-config/global-exceptions/exception",
127             "addExceptionConfig", "org.apache.struts.config.ExceptionConfig");
128 
129         digester.addRule("struts-config/global-exceptions/exception/set-property",
130             new BaseConfigSetPropertyRule());
131 
132         digester.addRule("struts-config/global-forwards",
133             new SetActionForwardClassRule());
134 
135         digester.addFactoryCreate("struts-config/global-forwards/forward",
136             new GlobalForwardFactory(cl));
137         digester.addSetProperties("struts-config/global-forwards/forward");
138         digester.addSetNext("struts-config/global-forwards/forward",
139             "addForwardConfig", "org.apache.struts.config.ForwardConfig");
140 
141         digester.addRule("struts-config/global-forwards/forward/set-property",
142             new BaseConfigSetPropertyRule());
143 
144         digester.addObjectCreate("struts-config/message-resources",
145             "org.apache.struts.config.MessageResourcesConfig", "className");
146         digester.addSetProperties("struts-config/message-resources");
147         digester.addSetNext("struts-config/message-resources",
148             "addMessageResourcesConfig",
149             "org.apache.struts.config.MessageResourcesConfig");
150 
151         digester.addRule("struts-config/message-resources/set-property",
152             new BaseConfigSetPropertyRule());
153 
154         digester.addObjectCreate("struts-config/plug-in",
155             "org.apache.struts.config.PlugInConfig");
156         digester.addSetProperties("struts-config/plug-in");
157         digester.addSetNext("struts-config/plug-in", "addPlugInConfig",
158             "org.apache.struts.config.PlugInConfig");
159 
160         digester.addRule("struts-config/plug-in/set-property",
161             new PlugInSetPropertyRule());
162 
163         // PluginConfig does not extend BaseConfig, at least for now.
164     }

168 /**
169  * <p> Class that records the name and value of a configuration property to be
170  * used in configuring a <code>PlugIn</code> instance when instantiated. </p>
171  */
172 final class PlugInSetPropertyRule extends Rule {
173     public PlugInSetPropertyRule() {
174         super();
175     }
176 
177     public void begin(String namespace, String names, Attributes attributes)
178         throws Exception {
179         PlugInConfig plugInConfig = (PlugInConfig) digester.peek();
180 
181         plugInConfig.addProperty(attributes.getValue("property"),
182             attributes.getValue("value"));
183     }
184 }

210 /**
211  * <p> A variant of the standard Digester <code>SetPropertyRule</code>.  If
212  * the element being processed has a "key" attribute, then the value will be
213  * used to call <code>setProperty(key,value)</code> on the object on top of
214  * the stack, which will be assumed to be of type <code>ActionConfig</code>.
215  * Otherwise, the standard <code>SetPropertyRule</code> behavior is invoked,
216  * and the value will be used to set a bean property on the object on top of
217  * the Digester stack. In that case, the element being processed is assumed to
218  * have attributes "property" and "value". </p>
219  */
220 final class BaseConfigSetPropertyRule extends SetPropertyRule {
221     public BaseConfigSetPropertyRule() {
222         super("property", "value");
223     }
224 
225     public void begin(Attributes attributes)
226         throws Exception {
227         if (attributes.getIndex("key") == -1) {
228             super.begin(attributes);
229 
230             return;
231         }
232 
233         if (attributes.getIndex("property") != -1) {
234             throw new IllegalArgumentException(
235                 "<set-property> accepts only one of 'key' or 'property' attributes.");
236         }
237 
238         Object topOfStack = digester.peek();
239 
240         if (topOfStack instanceof BaseConfig) {
241             BaseConfig config = (BaseConfig) topOfStack;
242 
243             config.setProperty(attributes.getValue("key"),
244                 attributes.getValue("value"));
245         } else {
246             throw new IllegalArgumentException(
247                 "'key' attribute of <set-property> only applicable to subclasses of BaseConfig; "
248                 + "object on top of stack is " + topOfStack + " [key: "
249                 + attributes.getValue("key") + ", value: "
250                 + attributes.getValue("value") + "]");
251         }
252     }
253 }

256 /**
257  * <p> An object creation factory which creates action form bean instances,
258  * taking into account the default class name, which may have been specified
259  * on the parent element and which is made available through the object on the
260  * top of the stack, which must be a <code>org.apache.struts.config.ModuleConfig</code>.
261  * </p>
262  */
263 final class ActionFormBeanFactory extends AbstractObjectCreationFactory {
264     private ClassLoader cl;
265 
266     public ActionFormBeanFactory(ClassLoader cl) {
267         super();
268         this.cl = cl;
269     }
270 
271     public Object createObject(Attributes attributes) {
272         // Identify the name of the class to instantiate
273         String className = attributes.getValue("className");
274 
275         if (className == null) {
276             ModuleConfig mc = (ModuleConfig) digester.peek();
277 
278             className = mc.getActionFormBeanClass();
279         }
280 
281         // Instantiate the new object and return it
282         Object actionFormBean = null;
283 
284         try {
285             actionFormBean = RequestUtils.applicationInstance(className, cl);
286         } catch (Exception e) {
287             digester.getLogger().error("ActionFormBeanFactory.createObject: ", e);
288         }
289 
290         return actionFormBean;
291     }
292 }
