22  package org.apache.struts.faces.taglib;
23  
24  import javax.el.ValueExpression;
25  import javax.faces.component.UIComponent;
26  
27  import org.apache.struts.faces.util.Utils;
28  
29  /**
30   * Custom tag that generates JavaScript for client side validation based
31   * on the validation rules loaded by the <code>ValidatorPlugIn</code>
32   * defined in the struts-config.xml file.  This is based on the code in
33   * the corresponding class of the Struts HTML tag library, modified as needed
34   * to reflect differences in the way JavaServer Faces renders field
35   * identifiers.
36   *
37   * @version $Rev$ $Date$
38   */
39  public class JavascriptValidatorTag extends AbstractFacesTag {
40  
41      // ---------------------------------------------------------- Tag Attributes
42  
43  
44      /**
45       * The name of the form that corresponds with the action name
46       * in struts-config.xml. Specifying a form name places a
47       * &lt;script&gt; &lt;/script&gt; around the JavaScript.
48       */
49      protected ValueExpression _formName;
50  
51      /**
52       * Sets the key (form name) that will be used
53       * to retrieve a set of validation rules to be
54       * performed on the bean passed in for validation.
55       * Specifying a form name places a &lt;script&gt;
56       * &lt;/script&gt; tag around the JavaScript.
57       */
58      public void setFormName(ValueExpression formName) {
59          this._formName = formName;
60      }
61  
62  
63      /**
64       * The current page number of a multi-part form.
65       * Only valid when the {@code formName} attribute is set.
66       */
67      protected ValueExpression _page;
68  
69      /**
70       * Sets the current page number of a multi-part form.
71       * Only field validations with a matching page number
72       * will be generated that match the current page number.
73       * Only valid when the {@code formName} attribute is set.
74       */
75      public void setPage(ValueExpression page) {
76          this._page = page;
77      }
78  
79  
80      /**
81       * This will be used as is for the JavaScript validation method name if it
82       * has a value. This is the method name of the main JavaScript method that
83       * the form calls to perform validations.
84       */
85      protected ValueExpression _method;
86  
87      /**
88       * Sets the method name that will be used for the JavaScript
89       * validation method name if it has a value.  This overrides
90       * the auto-generated method name based on the key (form name)
91       * passed in.
92       */
93      public void setMethod(ValueExpression method) {
94          this._method = method;
95      }
96  
97  
98      /**
99       * The static JavaScript methods will only be printed if this is set to
100      * {@code true}.
101      */
102     protected ValueExpression _staticJavascript;
103 
104     /**
105      * Sets whether or not to generate the static
106      * JavaScript.  If this is set to {@code true}, which
107      * is the default, the static JavaScript will be generated.
108      */
109     public void setStaticJavascript(ValueExpression staticJavascript) {
110         this._staticJavascript = staticJavascript;
111     }
112 
113 
114     /**
115      * The dynamic JavaScript objects will only be generated if this is set to
116      * {@code true}.
117      */
118     protected ValueExpression _dynamicJavascript;
119 
120     /**
121      * Sets whether or not to generate the dynamic
122      * JavaScript.  If this is set to {@code true}, which
123      * is the default, the dynamic JavaScript will be generated.
124      */
125     public void setDynamicJavascript(ValueExpression dynamicJavascript) {
126         this._dynamicJavascript = dynamicJavascript;
127     }
128 
129 
130     /**
131      * The {@code src} attribute for html script element (used to include an
132      * external script resource). The {@code src} attribute is only recognized
133      * when the {@code formName} attribute is specified.
134      */
135     protected ValueExpression _src;
136 
137     /**
138      * Sets the {@code src} attribute's value when defining
139      * the html script element. The {@code src} attribute is only recognized
140      * when the {@code formName} attribute is specified.
141      */
142     public void setSrc(ValueExpression src) {
143         this._src = src;
144     }
145 
146 
147     /**
148      * The JavaScript methods will enclosed with html comments if this is set to
149      * {@code true}.
150      */
151     protected ValueExpression _htmlComment;
152 
153     /**
154      * Sets whether or not to delimit the JavaScript with html comments.
155      * If this is set to {@code true}, which is the default, the
156      * HTML-Comment will be surround the JavaScript.
157      */
158     public void setHtmlComment(ValueExpression htmlComment) {
159         this._htmlComment = htmlComment;
160     }
161 
162 
163     /**
164      * Hide JavaScript methods in a CDATA section for XHTML when {@code true}.
165      */
166     protected ValueExpression _cdata;
167 
168     /**
169      * Sets the CDATA-status.
170      * @param cdata The CDATA to set
171      */
172     public void setCdata(ValueExpression cdata) {
173         this._cdata = cdata;
174     }
175 
176 
177     // ------------------------------------------------------------- Tag Methods
178 
179 
180     /**
181      * <p>Release any allocated resources.</p>
182      */
183     public void release() {
184 
185         super.release();
186         _formName = null;
187         _page = null;
188         _method = null;
189         _staticJavascript = null;
190         _dynamicJavascript = null;
191         _src = null;
192         _htmlComment = null;
193         _cdata = null;
194 
195     }
196 
197 
198     // ---------------------------------------------------------- Public Methods
199 
200 
201     /**
202      * Return the type of component to be created for this tag.
203      */
204     public String getComponentType() {
205 
206         return ("org.apache.struts.faces.JavascriptValidator");
207 
208     }
209 
210 
211     /**
212      * Return the {@code rendererType} to be used for rendering
213      * our component.
214      */
215     public String getRendererType() {
216 
217         return ("org.apache.struts.faces.JavascriptValidator");
218 
219     }
220 
221 
222     // ------------------------------------------------------- Protected Methods
223 
224 
225     /**
226      * Override attributes set on this tag instance.
227      *
228      * @param component Component whose attributes should be overridden
229      */
230     protected void setProperties(UIComponent component) {
231 
232         super.setProperties(component);
233         Utils.setStringProperty(component, "formName", _formName);
234         Utils.setIntegerProperty(component, "page", _page);
235         Utils.setStringProperty(component, "method", _method);
236         Utils.setBooleanProperty(component, "staticJavascript", _staticJavascript);
237         Utils.setBooleanProperty(component, "dynamicJavascript", _dynamicJavascript);
238         Utils.setStringProperty(component, "src", _src);
239         Utils.setBooleanProperty(component, "htmlComment", _htmlComment);
240         Utils.setBooleanProperty(component, "cdata", _cdata);
241 
242     }
243 }
