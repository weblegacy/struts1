21  package org.apache.struts.taglib.html;
22  
23  import static org.junit.jupiter.api.Assertions.assertEquals;
24  
25  import java.util.Locale;
26  
27  import org.apache.struts.mock.MockHttpServletRequest;
28  import org.apache.struts.mock.MockHttpServletResponse;
29  import org.apache.struts.mock.MockPageContext;
30  import org.apache.struts.mock.MockServletConfig;
31  import org.junit.jupiter.api.BeforeEach;
32  import org.junit.jupiter.api.Test;
33  
34  /**
35   * Unit tests for the {@link HtmlTag}.
36   */
37  public class TestHtmlTag {
38  
39      private MockServletConfig       config;
40      private MockHttpServletRequest  request;
41      private MockHttpServletResponse response;
42      private MockPageContext         pageContext;
43      private HtmlTag                 htmlTag;
44  
45      /**
46       * Set up mock objects.
47       */
48      @BeforeEach
49      public void setUp() {
50          config      = new MockServletConfig();
51          request     = new MockHttpServletRequest();
52          response    = new MockHttpServletResponse();
53          pageContext = new MockPageContext(config, request, response);
54          htmlTag     = new HtmlTag();
55          htmlTag.setPageContext(pageContext);
56      }
57  
58      /**
59       * Test the "lang" attribute with valid characters.
60       */
61      @Test
62      public void testValidLangTrue() {
63  
64          // switch to render "lang" attribute
65          htmlTag.setLang(true);
66  
67          // Render for Locale.US
68          request.setLocale(Locale.US);
69          assertEquals("<html lang=\"en-US\">", htmlTag.renderHtmlStartElement(), "render en_US");
70  
71          // Render for Locale.ENGLISH
72          request.setLocale(Locale.ENGLISH);
73          assertEquals("<html lang=\"en\">", htmlTag.renderHtmlStartElement(), "render en");
74  
75          // Test valid characters
76          request.setLocale(new Locale("abcd-efghijklmnopqrstuvwxyz", "ABCDEFGHIJKLM-NOPQRSTUVWXYZ", ""));
77          assertEquals("<html lang=\"abcd-efghijklmnopqrstuvwxyz-ABCDEFGHIJKLM-NOPQRSTUVWXYZ\">", htmlTag.renderHtmlStartElement(), "valid characters");
78  
79      }
80  
81      /**
82       * Test the "lang" attribute with valid characters.
83       */
84      @Test
85      public void testValidLangFalse() {
86  
87          // switch to NOT render "lang" attribute
88          htmlTag.setLang(false);
89  
90          // Ignore for Locale.US
91          request.setLocale(Locale.US);
92          assertEquals("<html>", htmlTag.renderHtmlStartElement(), "ignore en_US");
93  
94          // Ignore for Locale.ENGLISH
95          request.setLocale(Locale.ENGLISH);
96          assertEquals("<html>", htmlTag.renderHtmlStartElement(), "ignore en");
97  
98      }
99  
100     /**
101      * Test an invalid "language"
102      */
103     @Test
104     public void testInvalidLanguage() {
105 
106         // switch to render "lang" attribute
107         htmlTag.setLang(true);
108 
109         // make sure HtmlTag is setup to render "lang" using a valid value
110         request.setLocale(Locale.US);
111         assertEquals("<html lang=\"en-US\">", htmlTag.renderHtmlStartElement(), "check valid");
112 
113         // Test script injection
114         request.setLocale(new Locale("/><script>alert()</script>", "", ""));
115         assertEquals("<html>", htmlTag.renderHtmlStartElement(), "invalid <script>");
116 
117         // Test <
118         request.setLocale(new Locale("abc<def", "", ""));
119         assertEquals("<html>", htmlTag.renderHtmlStartElement(), "invalid LT");
120 
121         // Test >
122         request.setLocale(new Locale("abc>def", "", ""));
123         assertEquals("<html>", htmlTag.renderHtmlStartElement(), "invalid GT");
124 
125         // Test /
126         request.setLocale(new Locale("abc/def", "", ""));
127         assertEquals("<html>", htmlTag.renderHtmlStartElement(), "invalid SLASH");
128 
129         // Test &
130         request.setLocale(new Locale("abc&def", "", ""));
131         assertEquals("<html>", htmlTag.renderHtmlStartElement(), "invalid AMP");
132 
133     }
134

164         // Test &

168     }
