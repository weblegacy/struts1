21  package org.apache.struts.taglib.html;
22  
23  import org.apache.struts.taglib.TagUtils;
24  
25  import jakarta.servlet.jsp.JspException;
26  
27  /**
28   * Convenience base class for the various input tags for text fields.
29   *
30   * @version $Rev$ $Date: 2004-10-16 12:38:42 -0400 (Sat, 16 Oct 2004)
31   *          $
32   */
33  public abstract class BaseFieldTag extends BaseInputTag {
34      private static final long serialVersionUID = 6316491607452965571L;
35  
36      // ----------------------------------------------------- Instance Variables
37  
38      /**
39       * Comma-delimited list of content types that a server processing this
40       * form will handle correctly.  This property is defined only for the
41       * <code>file</code> tag, but is implemented here because it affects the
42       * rendered HTML of the corresponding &lt;input&gt; tag.
43       */
44      protected String accept = null;
45  
46      /**
47       * The "redisplay contents" flag (used only on <code>password</code>).
48       */
49      protected boolean redisplay = true;
50  
51      /**
52       * The type of input field represented by this tag (text, password, or
53       * hidden).
54       */
55      protected String type = null;
56  
57      public String getAccept() {
58          return (this.accept);
59      }
60  
61      public void setAccept(String accept) {
62          this.accept = accept;
63      }
64  
65      public boolean getRedisplay() {
66          return (this.redisplay);
67      }
68  
69      public void setRedisplay(boolean redisplay) {
70          this.redisplay = redisplay;
71      }
72  
73      // --------------------------------------------------------- Public Methods
74  
75      /**
76       * Generate the required input tag. <p> Support for indexed property since
77       * Struts 1.1
78       *
79       * @throws JspException if a JSP exception has occurred
80       */
81      public int doStartTag() throws JspException {
82          TagUtils.getInstance().write(this.pageContext, this.renderInputElement());
83  
84          return (EVAL_BODY_BUFFERED);
85      }
86  
87      /**
88       * Renders a fully formed &lt;input&gt; element.
89       *
90       * @throws JspException
91       * @since Struts 1.2
92       */
93      protected String renderInputElement()
94          throws JspException {
95          StringBuilder results = new StringBuilder("<input");
96  
97          prepareAttribute(results, "type", this.type);
98          prepareAttribute(results, "name", prepareName());
99          prepareAttribute(results, "accesskey", getAccesskey());
100         prepareAttribute(results, "accept", getAccept());
101         prepareAttribute(results, "maxlength", getMaxlength());
102         prepareAttribute(results, "size", getCols());
103         prepareAttribute(results, "tabindex", getTabindex());
104         prepareValue(results);
105         results.append(this.prepareEventHandlers());
106         results.append(this.prepareStyles());
107         if (!isXhtml()) {
108             prepareAttribute(results, "autocomplete", getAutocomplete());
109         }
110         prepareOtherAttributes(results);
111         results.append(this.getElementClose());
112 
113         return results.toString();
114     }
115 
116     /**
117      * Render the value element
118      *
119      * @param results The StringBuilder that output will be appended to.
120      */
121     protected void prepareValue(StringBuilder results)
122         throws JspException {
123         results.append(" value=\"");
124 
125         if (value != null) {
126             results.append(this.formatValue(value));
127         } else if (redisplay || !"password".equals(type)) {
128             Object value =
129                 TagUtils.getInstance().lookup(pageContext, name, property, null);
130 
131             results.append(this.formatValue(value));
132         }
133 
134         results.append('"');
135     }
136 
137     /**
138      * Return the given value as a formatted <code>String</code>.  This
139      * implementation escapes potentially harmful HTML characters.
140      *
141      * @param value The value to be formatted. <code>null</code> values will
142      *              be returned as the empty String "".
143      * @throws JspException if a JSP exception has occurred
144      * @since Struts 1.2
145      */
146     protected String formatValue(Object value)
147         throws JspException {
148         if (value == null) {
149             return "";
150         }
151 
152         return TagUtils.getInstance().filter(value.toString());
153     }
154 
155     /**
156      * Release any acquired resources.
157      */
158     public void release() {
159         super.release();
160         accept = null;
161         name = Constants.BEAN_KEY;
162         redisplay = true;
163     }
164 }
