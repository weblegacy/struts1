36  /**
37   * Renders an HTML <html> element with appropriate language attributes if
38   * there is a current Locale available in the user's session.
39   *
40   * @version $Rev$ $Date: 2005-08-21 19:08:45 -0400 (Sun, 21 Aug 2005)
41   *          $
42   */
43  public class HtmlTag extends TagSupport {
44      private static final long serialVersionUID = -4739352136106954707L;
45  
46      // ------------------------------------------------------------- Properties
47  
48      /**
49       * The {@code Log} instance for this class.
50       */
51      private transient final Logger log =
52          LoggerFactory.getLogger(HtmlTag.class);
53  
54      /**
55       * The message resources for this package.
56       */
57      protected static MessageResources messages =
58          MessageResources.getMessageResources(Constants.Package
59              + ".LocalStrings");

61      /**
62       * Are we rendering an xhtml page?
63       */
64      protected boolean xhtml = false;

67       * What version of XHTML is being rendered?
68       *
69       * @since Struts 1.4
70       */
71      private BigDecimal xhtmlVersion = TagUtils.XHTML_1_0;

74       * Are we rendering a lang attribute?
75       *
76       * @since Struts 1.2
77       */
78      protected boolean lang = false;

89          return this.xhtmlVersion.toString();

93          if (xhtmlVersion != null) {
94              this.xhtmlVersion = new BigDecimal(xhtmlVersion);
95          } else {
96              this.xhtmlVersion = TagUtils.XHTML_1_0;
97          }
98      }

118     /**
119      * Process the start of this tag.
120      *
121      * @throws JspException if a JSP exception has occurred
122      */
123     public int doStartTag() throws JspException {
124         TagUtils.getInstance().write(this.pageContext,
125             this.renderHtmlStartElement());
126 
127         return EVAL_BODY_INCLUDE;
128     }
129 
130     /**
131      * Renders an &lt;html&gt; element with appropriate language attributes.
132      *
133      * @since Struts 1.2
134      */
135     protected String renderHtmlStartElement() {
136         StringBuilder sb = new StringBuilder("<html");
137 
138         String language = null;
139         String country = "";
140 
141         Locale currentLocale =
142             TagUtils.getInstance().getUserLocale(pageContext, Globals.LOCALE_KEY);
143 
144         language = currentLocale.getLanguage();
145         country = currentLocale.getCountry();
146 
147         boolean validLanguage = isValidRfc2616(language);
148         boolean validCountry  = isValidRfc2616(country);
149 
150         // XHTML document conformance
151         if (this.xhtml) {
152             this.pageContext.setAttribute(Globals.XHTML_KEY, "true",
153                 PageContext.PAGE_SCOPE);
154             this.pageContext.setAttribute(Globals.XHTML_VERSION_KEY,
155                 xhtmlVersion, PageContext.REQUEST_SCOPE);
156 
157             // 1.1
158             if (xhtmlVersion.equals(TagUtils.XHTML_1_1)) {
159                 sb.append(" xmlns=\"http://www.w3.org/1999/xhtml\"");
160                 sb.append(" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"");
161                 sb.append(" xsi:schemaLocation=\"http://www.w3.org/MarkUp/SCHEMA/xhtml11.xsd\"");
162             }
163             // 2.0
164             else if (xhtmlVersion.equals(TagUtils.XHTML_2_0)) {
165                 sb.append(" xmlns=\"http://www.w3.org/2002/06/xhtml2/\"");
166                 sb.append(" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"");
167                 sb.append(" xsi:schemaLocation=\"http://www.w3.org/2002/06/xhtml2/ http://www.w3.org/MarkUp/SCHEMA/xhtml2.xsd\"");
168             }
169             // 5.0
170             else if (xhtmlVersion.equals(TagUtils.XHTML_5_0)) {
171                 sb.append(" xmlns=\"http://www.w3.org/1999/xhtml\"");
172             }
173             // 1.0/Default
174             else {
175                 if (!xhtmlVersion.equals(TagUtils.XHTML_1_0)) {
176                     log.warn("Defaulting to XHTML 1.0. Unknown version: {}", xhtmlVersion);
177                     xhtmlVersion = TagUtils.XHTML_1_0;
178                 }
179                 sb.append(" xmlns=\"http://www.w3.org/1999/xhtml\"");
180             }
181         }
182 
183         // If language is specified, output the attribute
184         // unless XHTML is version >= 1.1
185         if (this.lang && validLanguage) {
186             if (!this.xhtml || (xhtmlVersion.compareTo(TagUtils.XHTML_1_1) < 0)) {
187                 sb.append(" lang=\"");
188                 sb.append(language);
189 
190                 if (validCountry) {
191                     sb.append("-");
192                     sb.append(country);
193                 }
194 
195                 sb.append("\"");
196             }
197         }
198 
199         if (this.xhtml && validLanguage) {
200             sb.append(" xml:lang=\"");
201             sb.append(language);
202 
203             if (validCountry) {
204                 sb.append("-");
205                 sb.append(country);
206             }
207 
208             sb.append("\"");
209         }
210 
211         sb.append(">");
212 
213         return sb.toString();
214     }
215 
216     /**
217      * Process the end of this tag.
218      *
219      * @throws JspException if a JSP exception has occurred
220      */
221     public int doEndTag() throws JspException {
222         TagUtils.getInstance().write(pageContext, "</html>");
223 
224         // Evaluate the remainder of this page
225         return (EVAL_PAGE);
226     }

228     /**

231     public void release() {
232         this.xhtml = false;
233         this.lang = false;
234     }

236     /**
237      * Check whether the value contains valid characters for the
238      * "Accept-Language" header according to RFC 2616 (section 14.4).
239      *
240      * @param value The value to check
241      * @return <code>true</code> if valid, otherwise <code>false</code>
242      */
243     private boolean isValidRfc2616(String value) {
244         if (value == null || value.length() == 0) {
245             return false;
246         }
247         for (int i = 0; i < value.length(); i++) {
248             char c = value.charAt(i);
249 
250             if (!(Character.isLetter(c) || c == '-')) {
251                 return false;
252             }
253         }
254         return true;
255     }
