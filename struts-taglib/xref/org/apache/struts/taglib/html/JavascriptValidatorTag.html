65      /**
66       * A Comparator to use when sorting ValidatorAction objects.
67       */
68      private static final Comparator<ValidatorAction> actionComparator =
69          (va1, va2) -> {
70              if (((va1.getDepends() == null)
71                  || (va1.getDepends().length() == 0))
72                  && ((va2.getDepends() == null)
73                  || (va2.getDepends().length() == 0))) {
74                  return 0;
75              } else if (((va1.getDepends() != null)
76                  && (va1.getDepends().length() > 0))
77                  && ((va2.getDepends() == null)
78                  || (va2.getDepends().length() == 0))) {
79                  return 1;
80              } else if (((va1.getDepends() == null)
81                  || (va1.getDepends().length() == 0))
82                  && ((va2.getDepends() != null)
83                  && (va2.getDepends().length() > 0))) {
84                  return -1;
85              } else {
86                  return va1.getDependencyList().size()
87                  - va2.getDependencyList().size();
88              }
89          };

336     /**
337      * Render the JavaScript for to perform validations based on the form
338      * name.
339      *
340      * @throws JspException if a JSP exception has occurred
341      */
342     public int doStartTag() throws JspException {
343         JspWriter writer = pageContext.getOut();
344 
345         try {
346             writer.print(this.renderJavascript());
347         } catch (IOException e) {
348             throw new JspException(e.getMessage(), e);
349         }
350 
351         return EVAL_BODY_BUFFERED;
352     }

354     /**
355      * Returns fully rendered JavaScript.
356      *
357      * @since Struts 1.2
358      */
359     protected String renderJavascript()
360         throws JspException {
361         StringBuilder results = new StringBuilder();
362 
363         ModuleConfig config =
364             TagUtils.getInstance().getModuleConfig(pageContext);
365         ValidatorResources resources =
366             (ValidatorResources) pageContext.getAttribute(
367               ValidatorPlugIn.VALIDATOR_KEY
368                 + config.getPrefix(), PageContext.APPLICATION_SCOPE);
369 
370         if (resources == null) {
371             throw new JspException(
372                 "ValidatorResources not found in application scope under key \""
373                 + ValidatorPlugIn.VALIDATOR_KEY + config.getPrefix() + "\"");
374         }
375 
376         Locale locale =
377             TagUtils.getInstance().getUserLocale(this.pageContext, null);
378 
379         Form form = null;
380         if ("true".equalsIgnoreCase(dynamicJavascript)) {
381             form = resources.getForm(locale, formName);
382             if (form == null) {
383                 throw new JspException("No form found under '" + formName
384                     + "' in locale '" + locale
385                     + "'.  A form must be defined in the "
386                     + "Commons Validator configuration when "
387                     + "dynamicJavascript=\"true\" is set.");
388             }
389         }
390 
391         if (form != null) {
392             if ("true".equalsIgnoreCase(dynamicJavascript)) {
393                 results.append(this.createDynamicJavascript(config, resources,
394                         locale, form));
395             } else if ("true".equalsIgnoreCase(staticJavascript)) {
396                 results.append(this.renderStartElement());
397 
398                 if ("true".equalsIgnoreCase(htmlComment)) {
399                     results.append(HTML_BEGIN_COMMENT);
400                 }
401             }
402         }
403 
404         if ("true".equalsIgnoreCase(staticJavascript)) {
405             results.append(getJavascriptStaticMethods(resources));
406         }
407 
408         if ((form != null)
409             && ("true".equalsIgnoreCase(dynamicJavascript)
410             || "true".equalsIgnoreCase(staticJavascript))) {
411             results.append(getJavascriptEnd());
412         }
413 
414         return results.toString();
415     }

417     /**
418      * Generates the dynamic JavaScript for the form.
419      *
420      * @param config
421      * @param resources
422      * @param locale
423      * @param form
424      */
425     private String createDynamicJavascript(ModuleConfig config,
426         ValidatorResources resources, Locale locale, Form form)
427         throws JspException {
428         StringBuilder results = new StringBuilder();
429 
430         MessageResources messages =
431             TagUtils.getInstance().retrieveMessageResources(pageContext,
432                 bundle, true);
433 
434         HttpServletRequest request =
435             (HttpServletRequest) pageContext.getRequest();
436         ServletContext application = pageContext.getServletContext();
437 
438         List<ValidatorAction> actions = this.createActionList(resources, form);
439 
440         final String methods =
441             this.createMethods(actions, this.stopOnError(config));
442 
443         String formName = form.getName();
444 
445         jsFormName = formName;
446 
447         if (jsFormName.charAt(0) == '/') {
448             String mappingName =
449                 TagUtils.getInstance().getActionMappingName(jsFormName);
450             ActionMapping mapping =
451                 (ActionMapping) config.findActionConfig(mappingName);
452 
453             if (mapping == null) {
454                 JspException e =
455                     new JspException(messages.getMessage("formTag.mapping",
456                             mappingName));
457 
458                 pageContext.setAttribute(Globals.EXCEPTION_KEY, e,
459                     PageContext.REQUEST_SCOPE);
460                 throw e;
461             }
462 
463             jsFormName = mapping.getAttribute();
464         }
465 
466         results.append(this.getJavascriptBegin(methods));
467 
468         for (ValidatorAction va : actions) {
469             int jscriptVar = 0;
470             String functionName = null;
471 
472             if ((va.getJsFunctionName() != null)
473                 && (va.getJsFunctionName().length() > 0)) {
474                 functionName = va.getJsFunctionName();
475             } else {
476                 functionName = va.getName();
477             }
478 
479             results.append("    function " + jsFormName + "_" + functionName
480                 + " () { \n");
481 
482             for (Field field : form.getFields()) {
483 
484                 // Skip indexed fields for now until there is a good way to
485                 // handle error messages (and the length of the list (could
486                 // retrieve from scope?))
487                 if (field.isIndexed() || (field.getPage() != page)
488                     || !field.isDependency(va.getName())) {
489                     continue;
490                 }
491 
492                 String message =
493                     Resources.getMessage(application, request, messages,
494                         locale, va, field);
495 
496                 message = (message != null) ? message : "";
497 
498                 // prefix variable with 'a' to make it a legal identifier
499                 results.append("     this.a" + jscriptVar++ + " = new Array(\""
500                     + field.getKey() + "\", \"" + escapeQuotes(message)
501                     + "\", ");
502 
503                 results.append("new Function (\"varName\", \"");
504 
505                 Map<String, Var> vars = field.getVars();
506 
507                 // Loop through the field's variables.
508                 for (Map.Entry<String, Var> entry : vars.entrySet()) {
509                     String varName = entry.getKey();
510                     Var var = entry.getValue();
511                     String varValue =
512                         Resources.getVarValue(var, application, request, false);
513                     String jsType = var.getJsType();
514 
515                     // skip requiredif variables field, fieldIndexed, fieldTest,
516                     // fieldValue
517                     if (varName.startsWith("field")) {
518                         continue;
519                     }
520 
521                     String varValueEscaped = escapeJavascript(varValue);
522 
523                     if (Var.JSTYPE_INT.equalsIgnoreCase(jsType)) {
524                         results.append("this." + varName + "="
525                             + varValueEscaped + "; ");
526                     } else if (Var.JSTYPE_REGEXP.equalsIgnoreCase(jsType)) {
527                         results.append("this." + varName + "=/"
528                             + varValueEscaped + "/; ");
529                     } else if (Var.JSTYPE_STRING.equalsIgnoreCase(jsType)) {
530                         results.append("this." + varName + "='"
531                             + varValueEscaped + "'; ");
532 
533                         // So everyone using the latest format doesn't need to
534                         // change their xml files immediately.
535                     } else if ("mask".equalsIgnoreCase(varName)) {
536                         results.append("this." + varName + "=/"
537                             + varValueEscaped + "/; ");
538                     } else {
539                         results.append("this." + varName + "='"
540                             + varValueEscaped + "'; ");
541                     }
542                 }
543 
544                 results.append(" return this[varName];\"));\n");
545             }
546 
547             results.append("    } \n\n");
548         }
549 
550         return results.toString();
551     }

553     private String escapeQuotes(String in) {
554         if ((in == null) || (in.indexOf("\"") == -1)) {
555             return in;
556         }
557 
558         StringBuilder buffer = new StringBuilder();
559         StringTokenizer tokenizer = new StringTokenizer(in, "\"", true);
560 
561         while (tokenizer.hasMoreTokens()) {
562             String token = tokenizer.nextToken();
563 
564             if (token.equals("\"")) {
565                 buffer.append("\\");
566             }
567 
568             buffer.append(token);
569         }
570 
571         return buffer.toString();
572     }

574     /**
575      * <p>Backslash-escapes the following characters from the input string:
576      * &quot;, &apos;, \, \r, \n.</p>
577      *
578      * <p>This method escapes characters that will result in an invalid
579      * Javascript statement within the validator Javascript.</p>
580      *
581      * @param str The string to escape.
582      * @return The string <code>s</code> with each instance of a double quote,
583      *         single quote, backslash, carriage-return, or line feed escaped
584      *         with a leading backslash.
585      */
586     private String escapeJavascript(String str) {
587         if (str == null) {
588             return null;
589         }
590 
591         int length = str.length();
592 
593         if (length == 0) {
594             return str;
595         }
596 
597         // guess at how many chars we'll be adding...
598         StringBuilder out = new StringBuilder(length + 4);
599 
600         // run through the string escaping sensitive chars
601         for (int i = 0; i < length; i++) {
602             char c = str.charAt(i);
603 
604             if ((c == '"') || (c == '\'') || (c == '\\') || (c == '\n')
605                 || (c == '\r')) {
606                 out.append('\\');
607             }
608 
609             out.append(c);
610         }
611 
612         return out.toString();
613     }

615     /**
616      * Determines if validations should stop on an error.
617      *
618      * @param config The <code>ModuleConfig</code> used to lookup the
619      *               stopOnError setting.
620      * @return <code>true</code> if validations should stop on errors.
621      */
622     private boolean stopOnError(ModuleConfig config) {
623         Object stopOnErrorObj =
624             pageContext.getAttribute(ValidatorPlugIn.STOP_ON_ERROR_KEY + '.'
625                 + config.getPrefix(), PageContext.APPLICATION_SCOPE);
626 
627         boolean stopOnError = true;
628 
629         if (stopOnErrorObj instanceof Boolean) {
630             stopOnError = ((Boolean) stopOnErrorObj).booleanValue();
631         }
632 
633         return stopOnError;
634     }

636     /**
637      * Creates the JavaScript methods list from the given actions.
638      *
639      * @param actions     A List of ValidatorAction objects.
640      * @param stopOnError If true, behaves like released version of struts 1.1
641      *                    and stops after first error. If false, evaluates all
642      *                    validations.
643      * @return JavaScript methods.
644      */
645     private String createMethods(List<ValidatorAction> actions, boolean stopOnError) {
646         StringBuilder methods = new StringBuilder();
647         final String methodOperator = stopOnError ? " && " : " & ";
648 
649         for (ValidatorAction va : actions) {
650 
651             if (methods.length() > 0) {
652                 methods.append(methodOperator);
653             }
654 
655             methods.append(va.getMethod()).append("(form)");
656         }
657 
658         return methods.toString();
659     }

661     /**
662      * Get List of actions for the given Form.
663      *
664      * @param resources
665      * @param form
666      * @return A sorted List of ValidatorAction objects.
667      */
668     private List<ValidatorAction> createActionList(ValidatorResources resources, Form form) {
669         List<String> actionMethods = new ArrayList<>();
670 
671         for (Field field : form.getFields()) {
672 
673             for (String depends : field.getDependencyList()) {
674                 if ((depends != null) && !actionMethods.contains(depends)) {
675                     actionMethods.add(depends);
676                 }
677             }
678         }
679 
680         List<ValidatorAction> actions = new ArrayList<>();
681 
682         // Create list of ValidatorActions based on actionMethods
683         Iterator<String> iterator = actionMethods.iterator();
684 
685         while (iterator.hasNext()) {
686             String depends = iterator.next();
687             ValidatorAction va = resources.getValidatorAction(depends);
688 
689             // throw nicer NPE for easier debugging
690             if (va == null) {
691                 throw new NullPointerException("Depends string \"" + depends
692                     + "\" was not found in validator-rules.xml.");
693             }
694 
695             if ((va.getJavascript() != null)
696                 && (va.getJavascript().length() > 0)) {
697                 actions.add(va);
698             } else {
699                 iterator.remove();
700             }
701         }
702 
703         Collections.sort(actions, actionComparator);
704 
705         return actions;
706     }

725     /**
726      * Returns the opening script element and some initial javascript.
727      */
728     protected String getJavascriptBegin(String methods) {
729         StringBuilder sb = new StringBuilder();
730         String name = jsFormName.replace('/', '_'); // remove any '/' characters
731 
732         name =
733             jsFormName.substring(0, 1).toUpperCase()
734             + jsFormName.substring(1, jsFormName.length());
735 
736         sb.append(this.renderStartElement());
737 
738         if (this.isXhtml() && "true".equalsIgnoreCase(this.cdata)) {
739             sb.append("//<![CDATA[\r\n");
740         }
741 
742         if (!this.isXhtml() && "true".equals(htmlComment)) {
743             sb.append(HTML_BEGIN_COMMENT);
744         }
745 
746         sb.append("\n    var bCancel = false; \n\n");
747 
748         if ((methodName == null) || (methodName.length() == 0)) {
749             sb.append("    function validate" + name + "(form) { \n");
750         } else {
751             sb.append("    function " + methodName + "(form) { \n");
752         }
753 
754         sb.append("        if (bCancel) { \n");
755         sb.append("            return true; \n");
756         sb.append("        } else { \n");
757 
758         // Always return true if there aren't any Javascript validation methods
759         if ((methods == null) || (methods.length() == 0)) {
760             sb.append("            return true; \n");
761         } else {
762             sb.append("            var formValidationResult; \n");
763             sb.append("            formValidationResult = " + methods + "; \n");
764             if (methods.indexOf("&&") >= 0) {
765                 sb.append("            return (formValidationResult); \n");
766             } else {
767                 //Making Sure that Bitwise operator works:
768                 sb.append("            return (formValidationResult == 1); \n");
769             }
770         }
771         sb.append("        } \n");
772         sb.append("    } \n\n");
773 
774         return sb.toString();
775     }

777     protected String getJavascriptStaticMethods(ValidatorResources resources) {
778         StringBuilder sb = new StringBuilder();
779 
780         sb.append("\n\n");
781 
782         for (ValidatorAction va : resources.getValidatorActions().values()) {
783 
784             if (va != null) {
785                 String javascript = va.getJavascript();
786 
787                 if ((javascript != null) && (javascript.length() > 0)) {
788                     sb.append(javascript + "\n");
789                 }
790             }
791         }
792 
793         return sb.toString();
794     }

796     /**
797      * Returns the closing script element.
798      */
799     protected String getJavascriptEnd() {
800         StringBuilder sb = new StringBuilder();
801 
802         sb.append("\n");
803 
804         if (!this.isXhtml() && "true".equals(htmlComment)) {
805             sb.append(HTML_END_COMMENT);
806         }
807 
808         if (this.isXhtml() && "true".equalsIgnoreCase(this.cdata)) {
809             sb.append("//]]>\r\n");
810         }
811 
812         sb.append("</script>\n\n");
813 
814         return sb.toString();
815     }

817     /**
818      * Constructs the beginning &lt;script&gt; element depending on XHTML
819      * status.
820      *
821      * @since Struts 1.2
822      */
823     protected String renderStartElement() {
824         StringBuilder start =
825             new StringBuilder("<script type=\"text/javascript\"");
826 
827         // there is no language attribute in XHTML
828         if (!this.isXhtml() && this.scriptLanguage) {
829             start.append(" language=\"Javascript1.1\"");
830         }
831 
832         if (this.src != null) {
833             start.append(" src=\"" + src + "\"");
834         }
835 
836         start.append("> \n");
837 
838         return start.toString();
839     }
