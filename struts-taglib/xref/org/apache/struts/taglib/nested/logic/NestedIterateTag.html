21  package org.apache.struts.taglib.nested.logic;
22  
23  import org.apache.struts.taglib.logic.IterateTag;
24  import org.apache.struts.taglib.nested.NestedNameSupport;
25  import org.apache.struts.taglib.nested.NestedPropertyHelper;
26  
27  import javax.servlet.http.HttpServletRequest;
28  import javax.servlet.jsp.JspException;
29  
30  import java.util.Map;
31  
32  /**
33   * NestedIterateTag. Slightly more complex that the other extensions. This one
34   * has to yield a proper index property. Very taxing.
35   *
36   * @version $Rev$ $Date: 2004-10-16 12:38:42 -0400 (Sat, 16 Oct 2004)
37   *          $
38   * @since Struts 1.1
39   */
40  public class NestedIterateTag extends IterateTag implements NestedNameSupport {
41      private static final long serialVersionUID = -8879469068532412076L;
42  
43      // The current nesting
44      private String nesting = null;
45  
46      // original tag properties
47      private String originalName = null;
48      private String originalProperty = null;
49  
50      // original nesting environment
51      private String originalNesting = null;
52      private String originalNestingName = null;
53  
54      /**
55       * Overriding method of the heart of the matter. Gets the relative
56       * property and leaves the rest up to the original tag implementation.
57       * Sweet.
58       *
59       * @return int JSP continuation directive. This is in the hands of the
60       *         super class.
61       */
62      public int doStartTag() throws JspException {
63          // original values
64          originalName = getName();
65          originalProperty = getProperty();
66  
67          // set the ID to make the super tag happy
68          if ((id == null) || (id.trim().length() == 0)) {
69              id = property;
70          }
71  
72          // the request object
73          HttpServletRequest request =
74              (HttpServletRequest) pageContext.getRequest();
75  
76          // original nesting details
77          originalNesting = NestedPropertyHelper.getCurrentProperty(request);
78          originalNestingName =
79              NestedPropertyHelper.getCurrentName(request, this);
80  
81          // set the bean if it's been provided
82          // (the bean that's been provided! get it!?... nevermind)
83          if (getName() == null) {
84              // the qualified nesting value
85              nesting =
86                  NestedPropertyHelper.getAdjustedProperty(request, getProperty());
87          } else {
88              // it's just the property
89              nesting = getProperty();
90          }
91  
92          // set the properties
93          NestedPropertyHelper.setNestedProperties(request, this);
94  
95          // get the original result
96          int temp = super.doStartTag();
97  
98          // set the new reference (including the index etc)
99          NestedPropertyHelper.setName(request, getName());
100         NestedPropertyHelper.setProperty(request, deriveNestedProperty());
101 
102         // return the result
103         return temp;
104     }
105 
106     /**
107      * The only added property to the class. For use in proper nesting.
108      *
109      * @return String value of the property and the current index or mapping.
110      */
111     private String deriveNestedProperty() {
112         Object idObj = pageContext.getAttribute(id);
113 
114         if (idObj instanceof Map.Entry) {
115             return nesting + "(" + ((Map.Entry<?, ?>) idObj).getKey() + ")";
116         } else {
117             return nesting + "[" + this.getIndex() + "]";
118         }
119     }
120 
121     /**
122      * This is only overriden as the include reference will need it's index
123      * updated.
124      *
125      * @return int JSP continuation directive.
126      */
127     public int doAfterBody() throws JspException {
128         // store original result
129         int temp = super.doAfterBody();
130         HttpServletRequest request =
131             (HttpServletRequest) pageContext.getRequest();
132 
133         if (temp != SKIP_BODY) {
134             // set the new reference
135             NestedPropertyHelper.setProperty(request, deriveNestedProperty());
136         }
137 
138         // return super result
139         return temp;
140     }
141 
142     /**
143      * Complete the processing of the tag. The nested tags here will restore
144      * all the original value for the tag itself and the nesting context.
145      *
146      * @return int to describe the next step for the JSP processor
147      * @throws JspException for the bad things JSP's do
148      */
149     public int doEndTag() throws JspException {
150         // the super's thing
151         int i = super.doEndTag();
152 
153         // request
154         HttpServletRequest request =
155             (HttpServletRequest) pageContext.getRequest();
156 
157         // reset the original tag values
158         super.setName(originalName);
159         super.setProperty(originalProperty);
160 
161         // reset the original nesting values
162         if (originalNesting == null) {
163             NestedPropertyHelper.deleteReference(request);
164         } else {
165             NestedPropertyHelper.setProperty(request, originalNesting);
166             NestedPropertyHelper.setName(request, originalNestingName);
167         }
168 
169         // job done
170         return i;
171     }
172 
173     /**
174      * Release the tag's resources and reset the values.
175      */
176     public void release() {
177         // let the super release
178         super.release();
179 
180         // reset the original value place holders
181         originalName = null;
182         originalProperty = null;
183         originalNesting = null;
184         originalNestingName = null;
185     }
186 }
